async def initialize(self) -> None:
    """初始化 MCP 服务器连接"""
    
    # 第 1 步：检查 API Key
    settings = get_settings()
    if not settings.amap_api_key:
        raise ValueError("未配置高德地图 API Key")
    
    # 第 2 步：定义 MCP 服务器参数
    self._server_params = StdioServerParameters(
        command="npx",                                    # 用 npx 运行
        args=["-y", "@amap/amap-maps-mcp-server"],       # 下载并运行这个包
        env={
            "AMAP_MAPS_API_KEY": settings.amap_api_key,  # 传给 MCP 服务器
            **dict(os.environ)                            # 保留其他环境变量
        }
    )
    
    # 第 3 步：启动 MCP 服务器进程
    # stdio_client 启动一个子进程，通过 stdin/stdout 通信
    self._client_context = stdio_client(self._server_params)
    self._read, self._write = await self._client_context.__aenter__()
    
    # 第 4 步：建立 MCP 会话
    self._session_context = ClientSession(self._read, self._write)
    self._session = await self._session_context.__aenter__()
    
    # 第 5 步：握手协议
    await self._session.initialize()  # 确认协议版本
    
    # 第 6 步：获取工具列表
    raw_tools = await load_mcp_tools(self._session)
    
    # 第 7 步：包装工具（添加 schema）
    self._tools = wrap_mcp_tools(raw_tools)
    self._tools_dict = {tool.name: tool for tool in self._tools}

第 3 部分：清理资源（第 208-229 行）
async def cleanup(self) -> None:
    """清理 MCP 服务器资源"""
    
    if self._session_context:
        await self._session_context.__aexit__(None, None, None)
    
    if self._client_context:
        await self._client_context.__aexit__(None, None, None)
    
    # 清空所有属性
    self._session = None
    self._client_context = None
    self._session_context = None
    self._tools = []
    self._tools_dict = {}
作用：应用关闭时正确退出 MCP 子进程，避免僵尸进程。

def get_tools_by_names(self, names: List[str]) -> List[BaseTool]:
    """根据名称获取指定工具"""
    # 例：get_tools_by_names(["maps_weather", "maps_text_search"])
    return [self._tools_dict[name] for name in names if name in self._tools_dict]
def list_available_tools(self) -> List[str]:
    """列出所有可用工具名称"""
    return list(self._tools_dict.keys())
